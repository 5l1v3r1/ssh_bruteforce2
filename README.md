# ssh_bruteforcev
Simple ssh bruteforcer on Python3                                
# INSTALL                             
git clone https://github.com/zertmark/ssh_bruteforce2.git                               
cd ssh_bruteforce2 && chmod +x installer.sh                                                       
./installer.sh                                                                            
# RUN                                                
python3 ssh_bruteforcer.py (host) (wordlist) -U login -P port
